NAME
       Net::Rsh - perl client for Rsh protocol

SYNOPSIS
         use Net::Rsh;

         $a=Net::Rsh->new();

         $host="cisco.router.com";
         $local_user="root";
         $remote_user="root";
         $cmd="sh ru";

         @c=$a->rsh($host,$local_user,$remote_user,$cmd);

         print @c;

DESCRIPTION
         Rsh protocol requires that the program be
         run as root or that the program be setuid to root
