NAME
       hello - friendly greeting program

SYNOPSIS
       hello [OPTION]...

DESCRIPTION
       Print a friendly, customizable greeting.

       -t, --traditional
              use traditional greeting

       -g, --greeting=TEXT
              use TEXT as the greeting message

       --help display this help and exit

       --version
              output version information and exit
