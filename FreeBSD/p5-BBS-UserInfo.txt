NAME
       BBS::UserInfo - Base class of BBS::UserInfo::XXX

SYNOPSIS
           use BBS::UserInfo;

           my $bot = BBS::UserInfo->new('Ptt', 'server' => 'ptt.cc');
           $bot->connect();
           my $data = $bot->query('username');
           print($data->{'logintimes'});
