NAME
       Moose::Test - A Test Runner for the Moose test suite

SYNOPSIS
           # t/001-attr--nohook.t -- test bare class
           use Test::More tests => 1;
           use Moose::Test::Case;
           Moose::Test::Case->new->run_tests;

           # t/001-attr--immutable.t -- munge class before tests run
           use Test::More tests => 1;
           use Moose::Test::Case;
           Moose::Test::Case->new->run_tests(
               after_last_pm => sub {
                   Class->meta->make_immutable;
               },
           )

           # t/001-attr/Class.pm
           package Class;
           use Moose;
           has name => (isa => 'Str');

           # t/001-attr/string.t
           my $obj = Class->new(name => "Lloyd");
           ok($obj, "Constructor works when validating a string");

DESCRIPTION
       This module provides an abstraction over the Moose test cases such that
       it makes it easier for them to be re-used in different contexts.
