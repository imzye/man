NAME
       qudo - command to create database scheme

USAGE
         qudo [opts]

OPTIONS
           --rdbms      database kind [required] ex) mysql SQLite Pg
           --db         database name [required] ex) qudo_test
           --user       db user                  ex) root
           --pass       db password.
           --host       db host.
           --help       disp help.
           --dry_run    disp schema only.
           --use_innodb using innodb (optional)
