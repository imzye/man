NAME
       knet_addrtostr - Convert an address to a host name.


SYNOPSIS
       #include <libknet.h>

       int knet_addrtostr(
           const struct sockaddr_storage   *ss,
           socklen_t                        sslen,
           char                            *addr_buf,
           size_t                           addr_buf_size,
           char                            *port_buf,
           size_t                           port_buf_size
       );

DESCRIPTION
       knet_addrtostr

       ss - sockaddr_storage to convert

       sslen - len of the sockaddr_storage

       host - IPaddr/hostname where to store data (recommended size:
       KNET_MAX_HOST_LEN)

       port - port buffer where to store data (recommended size:
       KNET_MAX_PORT_LEN)

RETURN VALUE
       knet_strtoaddr returns same error codes as getnameinfo
