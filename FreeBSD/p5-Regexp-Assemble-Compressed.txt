NAME
       Regexp::Assemble::Compressed - Assemble more compressed Regular
       Expression

SYNOPSIS
        use Regexp::Assemble::Compressed;

        my $ra = Regexp::Assemble::Compressed->new;
        my @cctlds = qw(ma mc md me mf mg mh mk ml mm mn mo mp
                        mq mr ms mt mu mv mw mx my mz);
        for my $tld ( @cctlds ) {
            $ra->add( $tld );
        }
        print $ra->re; # prints m[ac-hk-z].
                       # Regexp::Assemble prints m[acdefghklmnopqrstuvwxyz]

DESCRIPTION
       Regexp::Assemble::Compressed is a subclass of Regexp::Assemble.  It
       assembles more compressed regular expressions.
