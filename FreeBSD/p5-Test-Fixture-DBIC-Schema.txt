NAME

       DBIx::Class::Schema

SYNOPSIS
         # in your t/*.t
         use Test::Fixture::DBIC::Schema;
         my $data = construct_fixture(
           schema  => $self->model,
           fixture => 'fixture.yaml',
         );

         - schema: Entry

             timestamp: 2008-02-23 23:22:58
