NAME
       day - print a formatted date

Synopsis
       Some of these may work.

         day today
         day tomorrow
         day yesterday
         day next th
         day next thursday
         day last wednesday

         day last week
         day this week
         day last month
         day last mo-fr

         day next first tu
         day next last th

         day 2 weeks away
         day 2 weeks from tomorrow

   print_date
         print_date(@stuff);

   string_dates
         string_dates(@stuff);

   make_date
         $date = make_date(@stuff);

   days_ago
         my $date = days_ago($number);

   next_last
       Will eventually do more.

         my @list = next_last('next|last', $date_obj, 'week');

   from
       Returns all of the days (inclusive) between $day1 and $day2 (or $now)
       if $day2 is omitted.

         from($now, $day1, to => $day2);
