NAME
       Tk::Enscript - a text-to-postscript converter using Tk::Canvas

SYNOPSIS
           use Tk::Enscript;

           enscript($top,
                    -text   => $text,
                    -media  => 'A4',
                    -output => "/tmp/bla.%d.ps",
           );
