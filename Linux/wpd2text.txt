NAME
       wpd2text - manual page for wpd2text 0.10.3

SYNOPSIS
       wpd2text [OPTION] FILE

DESCRIPTION
       `wpd2text' converts WordPerfect documents to plain text.

OPTIONS
       --info display document metadata instead of the text

       --help show this help message

       --password PASSWORD
              try to decrypt password protected document

       --version
              print version and exit

REPORTING BUGS
