NAME
       DBI::Mock - mock a DBI if we can't find the real one

SYNOPSIS
         use DBI::Mock;

         my $dbh = DBI::Mock->connect($data_source, $user, $pass, \%attr) or die $DBI::Mock::errstr;
         my $sth = $dbh->prepare();
         $sth->execute();
