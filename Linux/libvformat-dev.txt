NAME
       vf_is_modified - vf_is_modified()

SYNOPSIS
       #include <../vformat/vf_iface.h>

       extern bool_t vf_is_modified(VF_OBJECT_T *p_object);

PARAMETERS
       VF_OBJECT_T *p_object
              The object.

DESCRIPTION
       Fetch status of modified flag.

RETURNS
       TRUE iff modified since list read/write.
