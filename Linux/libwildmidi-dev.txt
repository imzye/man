NAME
       WildMidi_GetVersion - Get the library version

LIBRARY
       libWildMidi

SYNOPSIS
       #include <wildmidi_lib.h>

       long WildMidi_GetVersion (void);

DESCRIPTION
       Retrieve the library version.  Does not require library initialization.

RETURN VALUE
       Returns the library version

EXAMPLES
       long ver = WildMidi_GetVersion();
       printf("libWildMidi-%ld.%ld.%ld\n",
              (ver>>16) & 255, (ver>>8) & 255, (ver) & 255);
