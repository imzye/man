NAME
       Pass::OTP::URI - Parse otpauth:// URI

SYNOPSIS
           use Pass::OTP::URI qw(parse);

           my $uri = "otpauth://totp/ACME:john.doe@email.com?secret=HXDMVJECJJWSRB3HWIZR4IFUGFTMXBOZ&issuer=ACME&digits=6";
           my %options = parse($uri);

FUNCTIONS
       parse($uri)
