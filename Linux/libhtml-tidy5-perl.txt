NAME
       webtidy5 - HTML validation tool

SYNOPSIS
       webtidy [filename or url]... (filename - reads STDIN)

OPTIONS
       --context[=n]
              Show the offending line (and n surrounding lines)

       --noerrors
              Ignore errors

       --nowarnings
              Ignore warnings

       --help This message
