NAME
        mtx-plain - Plain TeX Runner

SYNOPSIS
       mtxrun --script plain [ OPTIONS ...  ] [ FILENAMES ]

DESCRIPTION
       Plain TeX Runner

OPTIONS
       --make create format file

       --fonts
              create plain font database

       --run  process file

       --format=string
              format name (default: luatex-plain)

       --engine=string
              engine to use (default: luatex)

       --jit  use luajittex
