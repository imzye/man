NAME
       Forest::Tree::Writer::ASCIIWithBranches - A slightly more complex ASCII writer

SYNOPSIS
         use Forest::Tree::Writer::ASCIIWithBranches;

         my $w = Forest::Tree::Writer::ASCIIWithBranches->new(tree => $tree);

         print $w->as_string; # outputs ....
         # root
         #    |---1.0
         #    |   |---1.1
         #    |   |---1.2
         #    |       |---1.2.1
         #    |---2.0
         #    |   |---2.1
         #    |---3.0
         #    |---4.0
         #        |---4.1
         #            |---4.1.1
