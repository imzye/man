NAME
       fflas-ffpack-config â€“ print details about the fflas-ffpack installation

SYNOPSIS
       fflas-ffpack-config [OPTION]

OPTIONS
       --prefix
              show FFLAS-FFPACK installation prefix

       --libs print library linking information

       --blas-libs
              print BLAS library linking information

       --cflags
              print pre-processor and compiler flags

       --blas-cflags
              print BLAS pre-processor and BLAS compiler flags

       --blas-home
              print where BLAS were found

       --help display this help and exit

       --version
              output version information string

       --decimal-version
              output version in decimal representation (base 100)
