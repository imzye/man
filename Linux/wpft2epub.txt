NAME
       wpft2epub - manual page for wpft2epub 0.9.6

SYNOPSIS
       wpft2epub [OPTIONS] INPUT OUTPUT

DESCRIPTION
       `wpft2epub' converts documents to ePub.

OPTIONS
       --help show this help message

       --version
              print version and exit

       --encoding ENCODING
              set the INPUT encoding. Use --list-encodings to see which encodings can be used.

       --list-encodings
              show the available encodings and exit

       --password PASSWORD
              set password to open the file
