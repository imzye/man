NAME
       rifiuti2 — Windows recycle bin analysis tool

SYNOPSIS
       rifiuti or rifiuti-vista [-h | -v | --help | --version]
       rifiuti [-l codepage] [-f xml | -f json | [-n] [-t delim]] [-z] [-o outfile] [--] info2_file
       rifiuti-vista [-f xml | -f json | [-n] [-t delim]] [-z] [-o outfile] [--] recycle_dir_or_file
       (for Windows and WSL) rifiuti-vista --live [-f xml | -f json | [-n] [-t delim]] [-z] [-o outfile]
