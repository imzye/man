NAME
       Catalyst::ActionRole::NoSSL - Force an action to be plain.

VERSION
       version 1.00

SYNOPSIS
         package MyApp::Controller::Foo;

         use parent qw/Catalyst::Controller::ActionRole/;

         sub bar : Local Does('RequireSSL') { ... }
         sub bar : Local Does('NoSSL') { ... }
